示例#

文本
const Demo = () => {
const [current, setCurrent] = React.useState(0)
const onClick = React.useCallback((e) => {
setCurrent(e.detail.value)
}, [])
return (
<TabBar
tabList={[{ title: '待办事项', text: 8 }, { title: '拍照' }, { title: '通讯录', dot: true }]}
current={current}
onClick={onClick}
/>
)
}
