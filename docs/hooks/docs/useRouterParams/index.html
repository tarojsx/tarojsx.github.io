useRouterParams

获取页面路由参数.

注意

参数值会被 decodeURIComponent.

用法#

import { useRouterParams } from '@tarojsx/hooks'
const Demo = () => {
const params = useRouterParams()
// pages/demo?title=abc
const title = useRouterParams('title')
// pages/demo?pageIndex=2
const pageNumber = useRouterParams('pageIndex', Number) + 1
// pageNumber === 3
}
